CREATE TABLE messages (
    id SERIAL PRIMARY KEY,
    дата TIMESTAMP,
    тип_сообщения VARCHAR(255),
    должник VARCHAR(255),
    должник_ссылка TEXT,
    арбитр VARCHAR(255),
    арбитр_ссылка TEXT,
    сообщение_ссылка TEXT,

    номер_сообщения VARCHAR(50),
    дата_публикации TIMESTAMP,

    наименование_должника VARCHAR(255),
    адрес TEXT,
    ОГРН VARCHAR(30),
    ИНН VARCHAR(20),
    номер_дела VARCHAR(50),
    дата_рождения DATE,
    место_рождения TEXT,
    место_жительства TEXT,
    СНИЛС VARCHAR(30),

    ФИО_АУ VARCHAR(300),
    адрес_корреспонденции TEXT,
    почта VARCHAR(255),
    СРО_АУ TEXT,
    адрес_СРО_АУ TEXT,
    объявление_о_проведении_торгов TEXT,

    торгова_площадка VARCHAR(255),
    номер_торгов VARCHAR(50),

    номер_лота INTEGER,
    описание TEXT,
    сведения_о_заключении_договора TEXT,
    номер_договора VARCHAR(50),
    дата_заключения_договора DATE,
    цена_приобретения_имущества NUMERIC(30, 2),
    наименование_покупателя VARCHAR(255),
    текст TEXT,

    классификация TEXT,

    ДКП TEXT,
    вид_торгов VARCHAR(255),
    дата_начала_подачи_заявок TIMESTAMP,
    дата_окончания_подачи_заявок TIMESTAMP,
    правила_подачи_заявок TEXT,
    дата_время_торгов TIMESTAMP,
    форма_подачи_предложения_о_цене TEXT,
    место_проведения TEXT,

    тип_оценки TEXT,
    дата_определения_стоимости DATE,
    балансовая_стоимость NUMERIC(30, 2)
);
